7.Course Performance Report (Cursor Logic):

FROM ExamResults er
JOIN Register r ON er.registration_id = r.id
WHERE r.course_id = 1;
=====================
SELECT status, COUNT(*)
FROM ExamResults er
JOIN Register r ON er.registration_id = r.id
WHERE r.course_id = 1
GROUP BY status;
